Interface BreedingRepoPort

繁殖リポジトリポート。

繁殖集約の永続化、イベント履歴、統計情報の取得を提供します。

interface BreedingRepoPort {
    appendBreedingEvent(
        cattleId: CattleId,
        event: BreedingEvent,
    ): Promise<void>;
    delete(cattleId: CattleId): Promise<void>;
    findByCattleId(cattleId: CattleId): Promise<null | BreedingAggregate>;
    findCattleNeedingAttention(
        ownerUserId: UserId,
        currentDate: Date,
    ): Promise<CattleId[]>;
    getBreedingHistory(
        cattleId: CattleId,
        startDate?: Date,
        endDate?: Date,
    ): Promise<BreedingEvent[]>;
    getBreedingStatistics(
        ownerUserId: UserId,
        startDate: Date,
        endDate: Date,
    ): Promise<
        {
            averagePregnancyRate: number;
            difficultBirthRate: number;
            totalCalvings: number;
            totalInseminations: number;
            totalPregnancies: number;
        },
    >;
    save(aggregate: BreedingAggregate): Promise<BreedingAggregate>;
    updateBreedingStatusDays(
        cattleId: CattleId,
        currentTime: Date,
    ): Promise<void>;
}
